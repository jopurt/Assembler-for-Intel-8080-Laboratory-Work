LXI H, 0x0026
MVI D, 10
MVI B, 0x0000        ; счетчик четных элементов
MVI C, 0x0000        ; счетчик нечетных элементов
MVI E, 0x0000	; счетчик номера


LOOP:
MOV A, E

ANA A          ; проверка на четность числа
JPE EVEN           ; если четное, переходим к метке EVEN

; Если число нечетное
MOV A,C
ADD M   
MOV C,A          ; увеличиваем значение нечетных чисел на 1
JMP CONTINUE

EVEN:
MOV A,B
ADD M             ; увеличиваем счетчик четных чисел на 1
MOV B,A
JMP CONTINUE

CONTINUE:
INX H             ; идем к следующему элементу
INR E
DCR D            
JNZ LOOP          ; если не конец массива, продолжаем


MOV M, C	; сохраняем количество нечетных чисел в текущей ячейке памяти

INX H

MOV M, B	; сохраняем количество четных чисел в следующей ячейке памяти

HLT
