LXI H, 0x0016 ; загрузить адрес x1 в регистры HL
MOV B, M ; загрузить значение в регистр B

INX H ; перейти к следующему адресу
MOV A, M  
CMA ; выполнить операцию НЕ (битовый инверс)
MOV C, A ; сохранить результат в регистре C

INX H  
MOV A, M 
CMA  
MOV D, A 

INX H 
MOV A, M 
XRA D ; выполнить операцию искоючающее или с x3 (уже инвертированным) и x4
MOV D, A ; сохранить результат в регистре D

ANA C ; выполнить операцию AND с x2 (уже инвертированным)
ORA D ; выполнить операцию OR с результатом XRA D
INX H
MOV M, A ; сохранить результат в памяти

HLT
